UBUNTU SERVER SETUP [tested on ubuntu server 18.04]
===================

UPDATE PKGS [login as root]
-----------
apt-get update
apt-get upgrade

SETUP ACCOUNT FOR APPLICATION
-----------------------------
adduser VisitorLog

INSTALL DOTNET CORE RUNTIME
---------------------------
wget https://packages.microsoft.com/config/ubuntu/20.04/packages-microsoft-prod.deb -O packages-microsoft-prod.deb
dpkg -i packages-microsoft-prod.deb
rm packages-microsoft-prod.deb
apt-get update
apt-get install -y apt-transport-https aspnetcore-runtime-3.1
dotnet --info

INSTALL MONGODB SERVER
----------------------
wget -qO - https://www.mongodb.org/static/pgp/server-4.4.asc | apt-key add -
echo "deb [ arch=amd64,arm64 ] https://repo.mongodb.org/apt/ubuntu focal/mongodb-org/4.4 multiverse" | tee /etc/apt/sources.list.d/mongodb-org-4.4.list
apt-get update
apt-get install -y mongodb-org
mkdir /mongodb
mkdir /mongodb/log
chown -R mongodb:mongodb /mongodb
echo "" > /etc/mongod.conf
nano /etc/mongod.conf
	[copy+paste from "mongod.conf"]
systemctl enable mongod.service
service mongod start
service mongod status
mongo --eval "rs.initiate()"

SETUP APP SECRETS
-----------------
nano /etc/environment
	[copy+paste from "env-variables.txt"]
	[set the correct secrets]
shutdown -r 0

DEPLOY .NET CORE APP [login with VisitorLog account]
--------------------
mkdir /home/VisitorLog/app
	[copy the files over via sFTP]
cd /home/VisitorLog/app
sudo chmod +x VisitorLog.dll
dotnet ./VisitorLog.dll
exit

INSTALL NGINX WITH LETSENCRYPT SSL [login as root]
----------------------------------
apt-get install -y software-properties-common certbot python3-certbot-nginx nginx
systemctl start nginx
systemctl status nginx
echo "" > /etc/nginx/sites-available/default
nano /etc/nginx/sites-available/default
	[copy+paste from "nginx.conf"]
certbot certonly -d VisitorLog.com -i nginx
certbot renew --dry-run
systemctl list-timers
nginx -t
systemctl restart nginx
systemctl status nginx

INSTALL KESTRAL SERVICE
-----------------------
nano /etc/systemd/system/VisitorLog.service
	[copy+paste from "VisitorLog.service"]
systemctl enable VisitorLog.service
systemctl start VisitorLog.service
systemctl status VisitorLog.service

CONFIGURE FIREWALL
------------------
apt-get install -y ufw
ufw reset
ufw allow 22
ufw allow 80
ufw allow 443
ufw deny from any
ufw enable
ufw status

RESTART
-------
shutdown -r 0
